# The vLLM base image already ships torch + a recent fastapi / uvicorn / httpx /
# pydantic. DO NOT pin those: glmocr requires pydantic>=2.12 (which the base
# already has), so a hard pin like pydantic==2.10.4 makes the resolver fail.
# Leave them unpinned so pip keeps the base versions and only adds the pipeline.
#
# Layout-aware OCR pipeline: PP-DocLayoutV3 (torch/transformers, not paddle) +
# GLM-OCR recognition. The [selfhosted] extra pulls torch/torchvision/
# transformers/opencv/pypdfium2.
glmocr[selfhosted]
# transformers >= 5.13 requires mistral-common >= 1.11.5 at import time
# (is_mistral_common_available version-gates); the vLLM base ships an older
# mistral-common, which makes vLLM's /v1/chat/completions 500 with
# "NameError: SpecialTokens is not defined". Pin the pair we verified working.
transformers==5.14.1
mistral-common[image]>=1.11.5
fastapi
uvicorn[standard]
httpx
